## Install

```bash
npm install --save ct-loader
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'ct-loader'
import 'ct-loader/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
